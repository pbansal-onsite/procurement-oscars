<script type="module">

import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-app.js";

import { getDatabase, ref, set, get, onValue, remove } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-database.js";


const firebaseConfig = {

  apiKey: "PASTE_HERE",

  authDomain: "PASTE_HERE",

  databaseURL: "PASTE_HERE",

  projectId: "PASTE_HERE",

  appId: "PASTE_HERE"

};


const app = initializeApp(firebaseConfig);

const db = getDatabase(app);


const SUPERLATIVES = [

  {icon:"🧾",label:"Most likely to negotiate a restaurant bill"},

  {icon:"🕵️",label:"Best poker face in a vendor meeting"},

  {icon:"📋",label:"Would create a RACI chart for a birthday party"},

  {icon:"💻",label:"Most tabs open at once"},

  {icon:"📨",label:"Sends 'just circling back' the fastest after no reply"},

  {icon:"🤝",label:"Most likely to be on a first-name basis with every supplier"},

  {icon:"🌙",label:"Would reconcile invoices for fun on a Friday night"},

  {icon:"💬",label:"The one everyone Slacks before sending anything externally"},

  {icon:"📱",label:"Most likely to have a supplier's personal cell number"},

  {icon:"🗂️",label:"Most likely to remember a PO number from 3 years ago"},

];


let timerInterval = null;

let currentIdx = 0;

let allVotes = {};

let voterCount = 0;


const voteUrl = window.location.origin + "/vote.html";

document.getElementById("vote-url").textContent = voteUrl;

document.getElementById("qr-img").src = `https://api.qrserver.com/v1/create-qr-code/?size=200x200&data=${encodeURIComponent(voteUrl)}`;


set(ref(db,"gameState"), {phase:"lobby", questionIdx:0});


onValue(ref(db,"voters"), snap => {

  voterCount = snap.exists() ? Object.keys(snap.val()).length : 0;

  document.getElementById("voter-count").textContent = voterCount;

  document.getElementById("q-total-voters").textContent = voterCount;

});


onValue(ref(db,"votes"), snap => {

  allVotes = snap.exists() ? snap.val() : {};

  if (document.getElementById("s-question").classList.contains("active")) {

    const v = allVotes[currentIdx] || {};

    const total = Object.values(v).reduce((a,b)=>a+b,0);

    document.getElementById("q-votes-in").textContent = total;

    document.getElementById("q-vote-count").textContent = total;

  }

});


window.startQuestion = function(idx) {

  currentIdx = idx;

  clearInterval(timerInterval);

  const s = SUPERLATIVES[idx];

  showScreen("s-question");

  document.getElementById("q-num").textContent = `Award ${idx+1} of ${SUPERLATIVES.length}`;

  document.getElementById("q-icon").textContent = s.icon;

  document.getElementById("q-label").textContent = s.label;

  document.getElementById("q-votes-in").textContent = "0";

  set(ref(db,"gameState"), {phase:"question", questionIdx:idx});

  let t = 10;

  const circle = document.getElementById("timer-circle");

  const ring = document.getElementById("timer-ring");

  circle.textContent = t;

  circle.classList.remove("hot");

  ring.classList.remove("pulse");

  timerInterval = setInterval(() => {

    t--;

    circle.textContent = t;

    if (t <= 3) { circle.classList.add("hot"); ring.classList.add("pulse"); }

    if (t <= 0) { clearInterval(timerInterval); window.showResults(); }

  }, 1000);

};


window.showResults = function() {

  clearInterval(timerInterval);

  const v = allVotes[currentIdx] || {};

  const entries = Object.entries(v).sort((a,b)=>b[1]-a[1]);

  const total = entries.reduce((s,e)=>s+e[1],0);

  const s = SUPERLATIVES[currentIdx];

  showScreen("s-results");

  set(ref(db,"gameState"), {phase:"results", questionIdx:currentIdx});

  document.getElementById("r-icon").textContent = s.icon;

  document.getElementById("r-label").textContent = s.label;

  const winnerName = entries[0] ? entries[0][0] : "No votes";

  const winnerVotes = entries[0] ? entries[0][1] : 0;

  document.getElementById("r-winner-name").textContent = winnerName;

  document.getElementById("r-winner-tally").textContent = `${winnerVotes} of ${total} vote${total!==1?"s":""}`;

  const barsEl = document.getElementById("r-bars");

  barsEl.innerHTML = "";

  entries.slice(0,6).forEach(([name,cnt],i) => {

    const pct = total > 0 ? Math.round((cnt/total)*100) : 0;

    const isW = i===0;

    const row = document.createElement("div");

    row.className = "bar-row fade-up";

    row.style.animationDelay = `${i*0.1}s`;

    row.innerHTML = `<div class="bar-label"><span class="name ${isW?"winner":""}">${isW?"🏆 ":""}${name}</span><span class="cnt">${cnt} vote${cnt!==1?"s":""} · ${pct}%</span></div><div class="bar-bg"><div class="bar-fill ${isW?"winner":"other"}" style="width:${pct}%"></div></div>`;

    barsEl.appendChild(row);

  });

  if (entries[0]) spawnConfetti("confetti-container");

  document.getElementById("btn-next-award").textContent = currentIdx < SUPERLATIVES.length-1 ? "Next award →" : "See leaderboard 🏆";

};


window.nextAward = function() {

  if (currentIdx < SUPERLATIVES.length-1) {

    window.startQuestion(currentIdx+1);

  } else {

    showScreen("s-leaderboard");

    set(ref(db,"gameState"), {phase:"leaderboard", questionIdx:currentIdx});

    renderLeaderboard();

    spawnConfetti("confetti-container2");

  }

};


function renderLeaderboard() {

  const tally = {};

  SUPERLATIVES.forEach((_,i) => {

    const v = allVotes[i] || {};

    const entries = Object.entries(v).sort((a,b)=>b[1]-a[1]);

    if (entries[0]) { const w = entries[0][0]; tally[w] = (tally[w]||0)+1; }

  });

  const sorted = Object.entries(tally).sort((a,b)=>b[1]-a[1]);

  const medals = ["🥇","🥈","🥉"];

  const cont = document.getElementById("lb-rows");

  cont.innerHTML = "";

  sorted.forEach(([name,cnt],i) => {

    const row = document.createElement("div");

    row.className = `lb-row${i===0?" top1":""}`;

    row.innerHTML = `<span class="lb-medal">${medals[i]||""}</span><span class="lb-name">${name}</span><span class="lb-awards">${cnt} award${cnt!==1?"s":""}</span>`;

    cont.appendChild(row);

  });

}


window.resetGame = function() {

  remove(ref(db,"votes"));

  remove(ref(db,"voters"));

  set(ref(db,"gameState"),{phase:"lobby",questionIdx:0});

  allVotes = {};

  showScreen("s-lobby");

};


function showScreen(id) {

  document.querySelectorAll(".screen").forEach(s=>s.classList.remove("active"));

  document.getElementById(id).classList.add("active");

}


function spawnConfetti(containerId) {

  const c = document.getElementById(containerId);

  c.innerHTML = "";

  const colors = ["#FF00BF","#fff","#a855f7","#22d3ee","#f59e0b"];

  for (let i=0;i<40;i++) {

    const p = document.createElement("div");

    p.className = "confetti-piece";

    const size = 6+Math.random()*8;

    p.style.cssText = `left:${Math.random()*100}%;top:0;width:${size}px;height:${size}px;background:${colors[Math.floor(Math.random()*colors.length)]};animation-delay:${Math.random()*1.5}s;`;

    c.appendChild(p);

  }

}
