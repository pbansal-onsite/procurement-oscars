<script type="module">

import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-app.js";

import { getDatabase, ref, set, onValue, runTransaction } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-database.js";


const firebaseConfig = {

  apiKey: "PASTE_HERE",

  authDomain: "PASTE_HERE",

  databaseURL: "PASTE_HERE",

  projectId: "PASTE_HERE",

  appId: "PASTE_HERE"

};


const app = initializeApp(firebaseConfig);

const db = getDatabase(app);


const SUPERLATIVES = [

  {icon:"🧾",label:"Most likely to negotiate a restaurant bill"},

  {icon:"🕵️",label:"Best poker face in a vendor meeting"},

  {icon:"📋",label:"Would create a RACI chart for a birthday party"},

  {icon:"💻",label:"Most tabs open at once"},

  {icon:"📨",label:"Sends 'just circling back' the fastest after no reply"},

  {icon:"🤝",label:"Most likely to be on a first-name basis with every supplier"},

  {icon:"🌙",label:"Would reconcile invoices for fun on a Friday night"},

  {icon:"💬",label:"The one everyone Slacks before sending anything externally"},

  {icon:"📱",label:"Most likely to have a supplier's personal cell number"},

  {icon:"🗂️",label:"Most likely to remember a PO number from 3 years ago"},

];


const NAMES = ["Angel Lin","Ashley Wood","Bradlee Dabu","Daniel Pappas","Dimitri Smith","Grace Nasr","Jan","Melissa Trager","Neha Kininge","Pooja Bansal","Rachel Goldstein","Raj Parikh","Rebecca Kim","Robert Lofthouse","Shradha Muktidoot","William Morris"];


const voterId = Math.random().toString(36).slice(2);

set(ref(db,`voters/${voterId}`), true);


let myVotes = {};

let lastPhase = "";

let lastIdx = -1;


function show(id) {

  document.querySelectorAll(".screen").forEach(s=>s.classList.remove("active"));

  document.getElementById(id).classList.add("active");

}


onValue(ref(db,"gameState"), snap => {

  if (!snap.exists()) return;

  const {phase, questionIdx} = snap.val();

  if (phase === lastPhase && questionIdx === lastIdx) return;

  lastPhase = phase; lastIdx = questionIdx;


  if (phase === "lobby") { show("s-lobby"); return; }

  if (phase === "leaderboard") { show("s-leaderboard"); return; }


  const s = SUPERLATIVES[questionIdx];


  if (phase === "question") {

    show("s-question");

    document.getElementById("q-num").textContent = `Award ${questionIdx+1} of ${SUPERLATIVES.length}`;

    document.getElementById("q-prog").style.width = Math.round(((questionIdx+1)/SUPERLATIVES.length)*100)+"%";

    document.getElementById("q-icon").textContent = s.icon;

    document.getElementById("q-label").textContent = s.label;

    const namesEl = document.getElementById("q-names");

    const votedWrap = document.getElementById("q-voted");

    if (myVotes[questionIdx]) {

      namesEl.style.display = "none";

      votedWrap.style.display = "flex";

      document.getElementById("q-voted-name").textContent = myVotes[questionIdx];

    } else {

      namesEl.style.display = "block";

      votedWrap.style.display = "none";

      namesEl.innerHTML = "";

      NAMES.forEach(name => {

        const btn = document.createElement("button");

        btn.className = "name-btn";

        btn.textContent = name;

        btn.onclick = () => {

          if (myVotes[questionIdx]) return;

          myVotes[questionIdx] = name;

          runTransaction(ref(db,`votes/${questionIdx}/${name}`), v => (v||0)+1);

          namesEl.style.display = "none";

          votedWrap.style.display = "flex";

          document.getElementById("q-voted-name").textContent = name;

        };

        namesEl.appendChild(btn);

      });

    }

  }


  if (phase === "results") {

    show("s-results");

    document.getElementById("r-icon").textContent = s.icon;

    document.getElementById("r-label").textContent = s.label;

    const myVote = myVotes[questionIdx];

    const myVoteEl = document.getElementById("r-my-vote");

    if (myVote) {

      myVoteEl.style.display = "block";

      document.getElementById("r-my-vote-name").textContent = myVote;

    } else {

      myVoteEl.style.display = "none";

    }

  }

});

</script>
